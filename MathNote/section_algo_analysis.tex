\subsection{算法复杂度的数学定义}

\begin{definition}
关于算法的复杂度本文使用如下定义

\begin{enumerate}
\item 如果存在正常数$c$和$n_0$，使得当$N \ge n_0$时， 都有$T(N) \le cf(N)$，则记为$ T(N) = \bigo(f(N))$。
\item 如果存在正常数$c$和$n_0$，使得当$N \ge n_0$时， 都有$T(N) \ge cg(N)$，则记为$ T(N) = \Om(g(N))$。
\item $ T(N) = \Th(h(N))$ 当且仅当$T(N) = \bigo(h(N))$且$T(N) = \Om(h(N))$。
\item 如果$ T(N) = \bigo(p(N))$ 且$T(N) \neq \Th(p(N))$，则$T(N) = (p(N))$。
\end{enumerate}
\end{definition}
